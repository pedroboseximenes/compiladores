class Good3 {
    public static void main(String[] args) {
        System.out.println(new Verificador().testa(5));
    }
}

class Verificador {
    public int testa(int valor) {
        int res;
        boolean flag;
        if (valor < 10) {
            res = 1;
            flag = true;
        } else {
            res = 0;
            flag = false;
        }
        return res;
    }
}

        System.out.println(new Loop().executa());

class Loop {
    public int executa() {
        int[] vetor;
        int i;
        int tam;
        tam = 5;
        vetor = new int[tam];
        i = 0;
        while (i < tam) {
            vetor[i] = i * 2;
            i = i + 1;
        }
        return vetor[4]; // Retorna o Ãºltimo elemento
    }
}

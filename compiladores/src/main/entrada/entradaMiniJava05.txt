class Good5 {
    public static void main(String[] args) {
        System.out.println(new Fatorial().compute(5));
    }
}

class Fatorial {
    public int compute(int num) {
        int num_aux;
        if (num < 1)
            num_aux = 1;
        else
            num_aux = num * (this.compute(num - 1));
        return num_aux;
    }
}
